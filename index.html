    function curl($url) 
{
  if ( empty($url) ) return false;

  $ch = curl_init($url);
  curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);

  $data = curl_exec($ch);
  curl_close($ch);

  return json_decode($data);
}
